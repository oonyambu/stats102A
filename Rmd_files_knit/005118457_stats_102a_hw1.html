#Below is the Pseudocode of the greatest_common_divisor():
**INPUT:** Input two integers a and b  
**OUTPUT:** The greatest common divisor of these two integers  

**IF** a < b, THEN
  temp <- a 
  a <- b 
  temp <- b 
**ELSE**
  remaining <- a %% b 
WHILE (remaining  0) 
  a = b 
  b= a 
  remaining <- a %% b 
ENDWHILE 
Set remaining = b 
return remaining
