1 : Greatest Common Divisor(GCD)

# pseudocode :
# 1. Set 'a', 'b' be maximum and minimum number
# 2. Set 'remainder' be numeric vector
# 3. Set 'remainder' be a remainder, where it is 'a' divided by 'b'
# 4. Switch 'b' to 'a' and 'remainder' to 'a'. Repeat from Step 3.
# 5. The algorithm stops when 'a' divided by 'b' is equal to 0. The remaining number contains the number which is a gcd number before zero number.
gcd <- function(x, y) {
  # this function takes two positive integers x, y and calculate their greatest common divisior (GCD) using the Euclidean algorithm
  # Args :
  # x: numeric variable type and 1 - dimension
  # y: numeric variable type and 1 - dimension
  # return: 
  # b : numeric variable type and 1 - dimension
  a <- max(x, y) 
  b <- min(x, y) 
  remainder <- numeric(1)
  
  if(sum(is.na(c(a, b))) > 0) {
    stop("this is NA!")
  }
  if(any(c(a, b) %% 1 != 0)) {
    stop("this is not integer!")
  }
  if(any(c(a, b) <= 0)) {
    stop("this is not a positive!")
  }
  if(sum(!is.numeric(c(a, b))) > 0) {
    stop("the input is not numeric!")
  }

  while(a %% b != 0) {
    remainder <- a %% b
    a <- b
    b <- remainder
  }
  b
}
gcd(1071, 462)
## [1] 21
gcd(462, 1071)
## [1] 21
gcd(646, 1406)
## [1] 38
gcd(1, 10)
## [1] 1
# pseudocode :
# 1. Sort x, y, z in regular sequence
# 2. Set three numbers be 'num' object from the smallest number to the biggiest number, which are a, b, c
# 3. Compare the smallest number and middle number and find the GCD number
# 4. Compare the GCD number from step 3 and the biggiest number. Find the GCD number.
# 5. Returns the GCD number from Step 4.
gcd_3 <- function(x, y, z) {
  # this function takes three positive integers x, y, z and returns their GCD
  # Args :
  # x: numeric variable type and 1 - dimension
  # y: numeric variable type and 1 - dimension
  # z: numeirc variable type and 1 - dimension
  # return: 
  # d : numeric variable type and 1 - dimension
  num <- sort(c(x, y, z))
  a <- num[1]
  b <- num[2]
  c <- num[3]
  
  r <- gcd(a, b)
  d <- gcd(r, c)
  d
}
gcd_3(1071, 462, 4452)
## [1] 21
gcd_3(646, 1406, 4218)
## [1] 38

2 : Prime Factorization

# pseudocode : 
# 1. Set 'prime' object be logical vector.
# 2. Start with a list a consecutive numbers 'k' from 2 to x - 1.
# 3. Set k = 2, the smallest numbers
# 4. Set 'prime' to be vectorized
# 5. Find TRUE or FALSE depending on a remainder is equal to 0
# 6. Set k to be the next number in the list. Repeat from Step 5.
# 7. The algorithm stops when there are no numbers left in the list larger than k. The remaining list contains all TRUE or FALSE up to 'x-1'
# 8. 'Prime' returns TRUE when the numbers in the list are all TRUE, otherwise returns FALSE 
is_prime <- function(x) {
  # This function returns TRUE or FALSE depending on whether or not the number x is prime
  # Args :
  # x: numeric variable type and 1 - dimension
  # return: 
  # prime : logical variable type and 1 - dimension
  prime <- logical()
  
  for(k in seq(2, length.out = x - 2)) {
   if(x %% k != 0) {
        prime <- c(prime, TRUE)
    } else {
        prime <- c(prime, FALSE)
    }
  }
  prime
  
  if(all(prime == TRUE)) {
    prime <- TRUE
  } else {
    prime <- FALSE
  }
  prime
}
is_prime(6)
## [1] FALSE
# pseudocode : 
# 1. Set 'factors' object be numeric vector.
# 2. Start with a list a consecutive numbers 'k' from 2 to x.
# 3. Set k = 2, the smallest numbers
# 4. Find a quotient of x divided by k and set a quotient be x
# 5. If remainder of x divided by k is equal to 0, repeat step 4
# 6. The algorithm stops when there are no numbers left in the list larger than k. The remaining list contains all factors up to x
get_factors <- function(x) {
  # this function takes a positive integer x and returns the vector of prime factors of x.
  # Args :
  # x: numeric variable type and 1 - dimension
  # return: 
  # factors : numeric variable type and 1 - dimension
  if (x <= 0) {
    stop("this is not a positive!")
  }
  if (x %% 1 != 0) {
    stop("this is not integer!")
  }

  factors <- numeric()
  for(k in seq(2, length.out = x - 1)) {
    if(x %% k == 0) {
      factors <- c(factors, k)
      x <- x %/% k
    }
    for(k in seq(2, length.out = x - 1)) {
      if(x %% k == 0) {
        factors <- c(factors, k)
        x <- x %/% k
      }
    }
  }
  factors
}
