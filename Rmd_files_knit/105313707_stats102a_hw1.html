gcd function:

gcd() Flowchart

Pseudocode
FUNCTION: gcd
INPUTS:   a (a positive integer); b (a positive integer)
OUTPUT:   an integer (numeric vector)

IF (a IS less than or equal to 0) OR IF (b IS less than or equal to 0)
  ERROR "The input must be positive integers, not 0 or negatives."
IF (a IS NOT an integer) OR IF (b IS NOT an integer)
  ERROR "The input must be positive integers, not doubles."

SET dividend <- maximum of a and b
SET divisor <- minimum of a and b

WHILE (divisor is NOT equal to 0) {
  SET remainder <- dividend MODULUS divisor
  SET dividend <- divisor
  SET divisor <- remainder

RETURN dividend

is_prime function:

is_prime Flowchart

Pseudocode
FUNCTION: is_prime_single
INPUTS:   num (a positive integer)
OUTPUT:   a logical value (TRUE OR FALSE)

SET prime <- NA
IF (num IS less than or equal to 1) OR (num IS NOT an integer)
  SET prime <- FALSE
ELSE
  FOR i in the sequence FROM 2 TO the (length of num - 2)
    IF num is divisible by i
      SET prime <- FALSE
      BREAK LOOP
  IF (prime IS NA) OR (num IS equal to 2)
    SET prime <- TRUE
RETURN prime

get_factors() Flowchart

FUNCTION: get_factors
INPUTS:   n (a positive integer)
OUTPUT:   a numerical vector

IF (num IS less than or equal to 0) OR (num IS NOT an integer)
  ERROR "The input must be a positive integer, not a negative or double."

SET factors <- c()

WHILE n IS divisible by 2
  SET factors <- c(factors, 2)
  SET n <- n / 2

FOR i in the sequence FROM 3 TO n BY increments of 2
  WHILE n IS divisible by i
    SET factors <- c(factors, i) 
    SET n <- n / i 

IF n IS prime
  SET factors <- c(factors, n) 

RETURN factors
