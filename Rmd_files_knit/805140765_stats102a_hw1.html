Pseudocode for gcd function:
GCD(a, b)
 IF (is.na(a) or is.na(b) )
   STOP
 IF (a <= 0 or b <= 0 )
   STOP
 IF (a %% 1 != 0 or b %% 1 != 0 )
   STOP
 remainder <- a %% b
 WHILE(remainder != 0)
   a <- b
   b <- remainder
   remainder <- a %% b
 RETURN b

Pseudocode for is_prime function:
IS_PRIME(x)
  IF (is.na(x))
    STOP
  result <- rep(TRUE, length(x))
  FOR (i in x)
    IF (i <= 1 | i %% 1 != 0)
      result[x==i] <- FALSE
    FOR (j in seq(2, length.out = i - 2))
      IF (i %% j == 0)
        result[x==i] <- FALSE
  RETURN result
is_prime(9)
## [1] FALSE
is_prime(c(2, 97, 23, 51, 119, -1, 0, 1.5))
## [1]  TRUE  TRUE  TRUE FALSE FALSE FALSE FALSE FALSE

Pseudocode for get_factors:
GET_FACTORS(x)
IF (is.na(x))
  STOP
IF (x <= 1)
  STOP
IF (x %% 1 != 0)
  STOP
factors <- integer(0)
WHILE(x != 1)
FOR (i in (2:x))
  IF (x %% i == 0)
    IF(is_prime(x))
      x <- x / i
      factors <- c(factors, i)
      break
RETURN factors
